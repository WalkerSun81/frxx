## 1.2 离线挂机机制

### 1.2.1 挂机前置条件

```python
def can_start_afk(role_id):
    role = get_role(role_id)
    
    # 1. 检查是否已在挂机
    if get_active_afk_record(role_id):
        return {'can': False, 'msg': '已在挂机中'}
    
    # 2. 检查上次离线时长
    last_afk = get_last_afk_record(role_id)
    if last_afk:
        offline_time = now() - last_afk['end_time']
        if offline_time < 300:  # 5分钟冷却
            return {'can': False, 'msg': '请等待5分钟后再挂机'}
    
    return {'can': True}
```

### 1.3.2 离线收益计算公式

```python
def calculate_offline_rewards(role_id, afk_record):
    role = get_role(role_id)
    map_id = afk_record['map_id']
    layer = afk_record['layer']
    
    # 1. 计算有效时长
    elapsed = now() - afk_record['start_time']
    max_time = VIP_AFK_LIMIT[role['vip_level']]  # VIP决定上限
    effective_time = min(elapsed, max_time)
    
    # 2. 修为收益
    cultivate_speed = calculate_cultivate_speed(role)
    exp_earned = int(cultivate_speed * effective_time * 0.8)  # 80%效率
    
    # 3. 灵石收益
    gold_per_second = calculate_gold_per_second(map_id, layer)
    gold_earned = int(gold_per_second * effective_time * 0.5)  # 50%效率
    
    # 4. 材料收益（基于击杀次数）
    avg_kill_time = get_avg_kill_time(role, map_id, layer)
    kill_count = int(effective_time / avg_kill_time)
    material_rate = get_material_drop_rate(map_id, layer)
    expected_materials = int(kill_count * material_rate * 0.3)  # 30%效率
    materials = roll_material_drops(expected_materials, map_id)
    
    # 5. 装备掉落（极低）
    equip_drop_rate = get_equip_drop_rate(map_id, layer, 'normal')
    expected_equips = int(kill_count * equip_drop_rate * 0.2)  # 20%效率
    equips = roll_equipment_drops(expected_equips, map_id, layer)
    
    return {
        'exp': exp_earned,
        'gold': gold_earned,
        'materials': materials,
        'equips': equips,
        'kill_count': kill_count,
        'effective_time': effective_time
    }
```

## 1.4 收益分段结算

### 1.4.1 分段机制

为了避免长时间离线收益过多，采用**指数衰减**机制：

```python
def calculate_decay_rewards(base_rewards, total_seconds):
    """
    分段衰减计算
    第1-2小时：100%效率
    第2-4小时：80%效率
    第4-8小时：50%效率
    第8小时+：25%效率
    """
    rewards = 0
    remaining = total_seconds
    
    # 第1段：2小时
    segment1 = min(remaining, 7200)
    rewards += base_rewards * (segment1 / 7200) * 1.0
    remaining -= segment1
    
    # 第2段：2小时
    if remaining > 0:
        segment2 = min(remaining, 7200)
        rewards += base_rewards * (segment2 / 7200) * 0.8
        remaining -= segment2
    
    # 第3段：4小时
    if remaining > 0:
        segment3 = min(remaining, 14400)
        rewards += base_rewards * (segment3 / 14400) * 0.5
        remaining -= segment3
    
    # 第4段：超过8小时
    if remaining > 0:
        rewards += base_rewards * (remaining / 14400) * 0.25
    
    return int(rewards)
```

## 1.5 离线收益领取

### 1.5.1 领取流程

```python
def claim_afk_rewards(role_id):
    # 1. 获取挂机记录
    afk_record = get_active_afk_record(role_id)
    if not afk_record:
        return {'success': False, 'msg': '没有进行中的挂机'}
    
    # 2. 计算实际收益
    offline_seconds = now() - afk_record['start_time']
    rewards = calculate_offline_rewards(role_id, afk_record)
    
    # 3. 应用衰减
    exp_with_decay = calculate_decay_rewards(
        rewards['exp'] / offline_seconds * 3600,
        offline_seconds
    )
    rewards['exp'] = exp_with_decay
    
    # 4. 发放奖励
    add_exp(role_id, rewards['exp'])
    add_gold(role_id, rewards['gold'])
    add_items(role_id, rewards['materials'])
    add_equips(role_id, rewards['equips'])
    
    # 5. 关闭挂机记录
    close_afk_record(afk_record)
    
    # 6. 记录日志
    log_afk_claim(role_id, rewards)
    
    return {'success': True, 'rewards': rewards}
```

### 1.7.2 定时结算器

```python
class AfkSettleJob:
    """
    每5分钟执行一次离线挂机结算
    避免玩家上线时一次性计算大量数据
    """
    interval = 300  # 5分钟
    
    def run(self):
        active_afks = get_active_afk_records()
        
        for afk in active_afks:
            # 跳过上次结算后不足5分钟的
            if now() - afk['last_settle_time'] < 300:
                continue
            
            # 计算这5分钟的收益
            rewards = calculate_afk_rewards(afk, 300)
            
            # 累加到玩家账户
            add_rewards_to_role(afk['role_id'], rewards)
            
            # 更新结算时间
            update_afk_record(afk, now())
```
