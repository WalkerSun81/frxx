```python
def generate_equipment_equip(base_attrs, quality):
    equip = {
        'base_attrs': base_attrs,
        'extra_attrs': [],
        'gem_holes': []
    }
    
    # 生成额外词条
    extra_count = get_extra_count(quality)  # 白0，绿1，蓝2...
    for i in range(extra_count):
        attr_type = weighted_random(ATTR_WEIGHTS)
        value = get_random_value(attr_type, base_level)
        equip['extra_attrs'].append({
            'type': attr_type,
            'value': value
        })
    
    # 生成宝石孔
    hole_count = get_hole_count(quality, realm_level)
    for i in range(hole_count):
        equip['gem_holes'].append({'mounted': None})
    
    return equip
```

### 1.5.3 强化算法

```python
def enhance_equipment(equip_id, use_protection):
    equip = get_equipment(equip_id)
    current_level = equip['level']
    target_level = current_level + 1
    
    # 检查强化上限
    max_level = get_max_enhance_level(equip['quality'])
    if current_level >= max_level:
        return {'success': False, 'msg': '已达强化上限'}
    
    # 检查消耗
    if not check_resources(['enhance_stone', 'copper'], get_cost(target_level)):
        return {'success': False, 'msg': '资源不足'}
    
    # 消耗资源
    deduct_resources(['enhance_stone', 'copper'], get_cost(target_level))
    
    # 成功率判定
    success_rate = get_success_rate(target_level)
    if random() < success_rate:
        # 成功
        new_level = target_level
        equip['level'] = new_level
        save_equipment(equip)
        return {'success': True, 'new_level': new_level}
    else:
        # 失败
        if use_protection:
            return {'success': False, 'msg': '强化失败，保护符生效'}
        
        penalty = get_penalty(target_level)
        new_level = max(0, current_level - penalty)
        equip['level'] = new_level
        save_equipment(equip)
        return {'success': False, 'new_level': new_level, 'penalty': penalty}
```

```python
def generate_drop(monster_type, map_level):
    # 1. 判定是否掉落装备（70%概率）
    if random() > 0.7:
        return {'has_equip': False}
    
    # 2. 品质抽取
    quality = roll_quality(monster_type)
    
    # 3. 装备类型抽取
    slot_type = random_slot_type()
    
    # 4. 装备等级（基于地图等级±2）
    equip_level = clamp(map_level + random(-2, 2), 1, MAX_LEVEL)
    
    # 5. 生成装备属性
    base_attrs = get_base_attrs(slot_type, equip_level)
    equip = generate_equipment_equip(base_attrs, quality)
    equip['slot_type'] = slot_type
    equip['level'] = equip_level
    equip['quality'] = quality
    
    return {'has_equip': True, 'equipment': equip}
```

```python
def calculate_equip_power(equip):
    power = 0
    # 基础属性战力
    for attr, value in equip['base_attrs'].items():
        power += value * ATTR_POWER_WEIGHT[attr]
    # 词条战力
    for attr in equip['extra_attrs']:
        power += attr['value'] * ATTR_POWER_WEIGHT[attr['type']]
    # 强化加成
    power *= (1 + equip['enhance_bonus'] / 100)
    # 宝石战力
    for hole in equip['gem_holes']:
        if hole['gem_id']:
            gem = get_gem(hole['gem_id'])
            power += calculate_gem_power(gem)
    return int(power)
```
