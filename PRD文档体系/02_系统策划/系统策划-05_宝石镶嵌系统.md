## 1.4 宝石属性计算公式

```python
def calculate_gem_attr(gem_type, grade):
    base_value = GEM_BASE_ATTR[gem_type]  # 1级基础值
    multiplier = GEM_GRADE_MULTIPLIER[grade]  # 品级倍率
    random_factor = random.uniform(0.9, 1.1)  # ±10%浮动
    
    final_value = base_value * multiplier * random_factor
    return round(final_value, 1)

# 示例：3级力量宝石
# base_value = 3, multiplier = 5.0
# final_value = 3 × 5.0 × 0.95 ~ 1.1 = 14.25 ~ 15
```

## 1.5 宝石合成算法

```python
def compose_gems(gem_ids, use_aid=False):
    # 1. 验证宝石
    gems = get_gems(gem_ids)
    if len(gems) != 2 or gems[0]['type'] != gems[1]['type']:
        return {'success': False, 'msg': '宝石类型不匹配'}
    
    if gems[0]['grade'] != gems[1]['grade']:
        return {'success': False, 'msg': '宝石品级不匹配'}
    
    grade = gems[0]['grade']
    gem_type = gems[0]['type']
    
    # 2. 检查目标品级
    target_grade = grade + 1
    if target_grade > 6:
        return {'success': False, 'msg': '已达最高品级'}
    
    # 3. 消耗判定
    if target_grade >= 4:
        aid_cost = COMPOSE_AID_COST[target_grade]  # 凝晶石数量
        if use_aid:
            if not check_resource('aid_stone', aid_cost):
                return {'success': False, 'msg': '凝晶石不足'}
            deduct_resource('aid_stone', aid_cost)
        else:
            # 无辅助材料，成功率降低
            success_rate = COMPOSE_NO_AID_RATE[target_grade]
            if random() > success_rate:
                # 失败，返还1个低级宝石
                return_gem(gems[0])
                return {'success': False, 'msg': '合成失败', 'returned': gems[0]}
    
    # 4. 销毁原料
    for gem_id in gem_ids:
        destroy_gem(gem_id)
    
    # 5. 生成新宝石
    new_gem = create_gem(gem_type, target_grade)
    
    return {'success': True, 'new_gem': new_gem}
```

### 1.6.2 宝石镶嵌规则

```python
def mount_gem(equipment_id, gem_id, hole_index):
    equip = get_equipment(equipment_id)
    gem = get_gem(gem_id)
    
    # 1. 验证孔位
    if hole_index >= len(equip['gem_holes']):
        return {'success': False, 'msg': '孔位不存在'}
    
    # 2. 验证孔位状态
    if equip['gem_holes'][hole_index]['gem_id'] is not None:
        return {'success': False, 'msg': '孔位已被占用'}
    
    # 3. 验证宝石所有权
    if gem['owner_type'] != 1 or gem['owner_id'] != current_role_id:
        return {'success': False, 'msg': '宝石不属于你'}
    
    # 4. 执行镶嵌
    equip['gem_holes'][hole_index]['gem_id'] = gem_id
    gem['owner_type'] = 2  # 装备持有
    gem['owner_id'] = equipment_id
    gem['slot_index'] = hole_index
    
    save_equipment(equip)
    save_gem(gem)
    
    return {'success': True}
```

```python
# 宝石属性汇总计算
def calculate_gem_bonus(role):
    total_bonus = {
        'strength': 0,
        'agility': 0,
        'vitality': 0,
        'intelligence': 0,
        'accuracy': 0,
        'sense': 0,
        'wisdom': 0,
        'luck': 0,
        'crit_rate': 0,
        'dodge_rate': 0,
        'fire_resist': 0,
        'water_resist': 0,
        # ...
    }
    
    # 遍历所有已镶嵌宝石
    for equip in get_equipped_items(role):
        for hole in equip['gem_holes']:
            if hole['gem_id']:
                gem = get_gem(hole['gem_id'])
                attr_type = GEM_TYPE_TO_ATTR[gem['type']]
                attr_value = calculate_gem_attr(gem['type'], gem['grade'])
                total_bonus[attr_type] += attr_value
    
    return total_bonus
```
