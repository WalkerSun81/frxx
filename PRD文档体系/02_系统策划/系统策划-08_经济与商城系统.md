    base_income = 0
    
    # 1. 挂机收益（60%）
    afk_speed = calculate_cultivate_speed(role) * 0.5
    # 灵石/修为转换比例1:1000
    afk_gold = afk_speed * 24 * 3600 / 1000
    base_income += afk_gold * 0.6
    
    # 2. 战斗掉落（25%）
    battle_gold = calculate_battle_income(role) * 0.25
    base_income += battle_gold
    
    # 3. 任务奖励（10%）
    task_gold = calculate_task_income(role) * 0.1
    base_income += task_gold
    
    # 4. 活动/其他（5%）
    base_income += 100  # 固定100
    
    # 5. 气运加成
    luck_multiplier = 1 + role['luck'] * 0.01
    
    return int(base_income * luck_multiplier)

## 1.5 购买算法

```python
def purchase_item(role_id, shop_type, item_id, count=1):
    role = get_role(role_id)
    
    # 1. 获取商品信息
    item = get_shop_item(shop_type, item_id)
    
    # 2. 检查限购
    if item['daily_limit'] > 0:
        purchased = get_daily_purchase_count(role_id, shop_type, item_id)
        if purchased + count > item['daily_limit']:
            return {'success': False, 'msg': '已达每日购买上限'}
    
    # 3. 扣除货币
    total_cost = item['price'] * count
    if not deduct_currency(role_id, item['currency'], total_cost):
        return {'success': False, 'msg': '货币不足'}
    
    # 4. 发放物品
    items = []
    for _ in range(count):
        item_data = generate_purchase_item(item)
        add_to_bag(role_id, item_data)
        items.append(item_data)
    
    # 5. 记录购买
    record_purchase(role_id, shop_type, item_id, count)
    
    return {'success': True, 'items': items}

def generate_purchase_item(item):
    """生成购买物品（支持随机属性）"""
    result = {
        'item_id': item['item_id'],
        'count': item['count'],
        'bind': item.get('bind', True)
    }
    
    # 如果是随机属性物品
    if item.get('random_attrs'):
        result['attrs'] = roll_random_attrs(item['random_attrs'])
    
    return result
```

def calculate_consumption_pressure(role):
    # 强化所需
    equip_cost = sum([get_enhance_cost(equip['level'] + 1) for equip in role['equipment']])
    
    # 突破所需
    break_cost = get_realm_break_cost(role['realm_level'])
    
    # 宝石所需
    gem_cost = calculate_gem_upgrade_cost(role)
    
    # 总压力
    total_weekly_consumption = equip_cost + break_cost * 0.2 + gem_cost * 0.3
    
    # 与收入对比
    weekly_income = calculate_weekly_income(role)
    
    return {
        'consumption': total_weekly_consumption,
        'income': weekly_income,
        'ratio': total_weekly_consumption / weekly_income if weekly_income > 0 else 999
    }

def get_enhance_cost(current_level):
    """获取强化消耗"""
    base_cost = {
        1: 100,
        2: 200,
        3: 400,
        4: 800,
        5: 1500,
        6: 3000,
        7: 6000,
        8: 12000,
        9: 25000,
        10: 50000
    }
    return base_cost.get(current_level, 100000)

## 1.10 经济平衡调整机制

### 1.10.1 监控指标

| 指标 | 正常范围 | 预警阈值 | 处理方案 |
|:---|:---:|:---:|:---|
| 灵石通胀率 | <5%/周 | >10%/周 | 增加消耗水槽 |
| 贫富差距 | <3倍 | >5倍 | 增加免费产出 |
| 付费转化率 | >3% | <2% | 优化付费点 |
| 留存率 | >40% | <30% | 调整前期体验 |

### 1.10.2 动态调整

- 每周分析经济数据
- 根据玩家反馈调整消耗/产出
- 大版本更新时重新评估经济模型
- 重大调整需经过小范围测试

### 1.10.3 玩家反馈渠道

- 游戏内客服系统
- 官方论坛/社区
- 问卷调查
- 玩家座谈会
