```python
def calculate_monster_attrs(map_id, layer, monster_type='normal'):
    base_level = get_map_base_level(map_id) + (layer - 1)
    
    # 等级缩放
    level = clamp(base_level, 1, 200)
    
    # 数值增长系数
    growth = 1 + (level - 1) * 0.04
    
    # 基础属性（练气期）
    base_attrs = {
        'hp': 50 * growth,
        'atk': 10 * growth,
        'def': 5 * growth,
        'speed': 8 * growth,
    }
    
    if monster_type == 'elite':
        # 精英怪属性×2.5
        base_attrs = {k: v * 2.5 for k, v in base_attrs.items()}
    elif monster_type == 'boss':
        # Boss属性×15
        base_attrs = {k: v * 15 for k, v in base_attrs.items()}
    
    return base_attrs
```

## 1.4 战斗流程

### 1.4.1 回合制战斗逻辑

```python
def battle_process(role, monster):
    # 1. 初始化
    role_hp = role['max_hp']
    monster_hp = monster['hp']
    round_count = 0
    battle_log = []
    
    # 2. 速度排序（先手）
    if role['speed'] > monster['speed']:
        first_attacker = 'role'
    else:
        first_attacker = 'monster'
    
    # 3. 战斗循环
    while role_hp > 0 and monster_hp > 0 and round_count < 50:
        round_count += 1
        
        # 回合开始
        attacker, defender = get_attacker_order(first_attacker, round_count)
        
        if attacker == 'role':
            # 玩家攻击
            damage = calculate_damage(role, monster)
            monster_hp -= damage
            battle_log.append(f"第{round_count}回合：你造成{damage}点伤害")
        else:
            # 怪物攻击
            damage = calculate_damage(monster, role)
            role_hp -= damage
            battle_log.append(f"第{round_count}回合：怪物造成{damage}点伤害")
        
        # 切换攻击方
        first_attacker = 'monster' if first_attacker == 'role' else 'role'
    
    # 4. 战斗结果
    if monster_hp <= 0:
        return {'result': 'win', 'rewards': calculate_rewards(role, monster, round_count)}
    elif role_hp <= 0:
        return {'result': 'lose', 'remaining_hp': 0}
    else:
        return {'result': 'timeout', 'remaining_hp': role_hp}
```

### 1.4.2 伤害计算公式

```python
def calculate_damage(attacker, defender):
    # 1. 基础伤害
    base_damage = attacker['atk']
    
    # 2. 暴击判定
    crit_rate = calculate_crit_rate(attacker)
    is_crit = random() < crit_rate
    if is_crit:
        crit_multiplier = 1.5 + attacker['luck'] * 0.02
        base_damage *= crit_multiplier
    
    # 3. 防御减免
    defense = defender['def']
    damage_reduction = defense / (defense + 500)
    final_damage = base_damage * (1 - damage_reduction)
    
    # 4. 随机浮动
    final_damage *= random.uniform(0.9, 1.1)
    
    # 5. 最低保底
    final_damage = max(final_damage, attacker['atk'] * 0.1)
    
    return int(final_damage)

def calculate_crit_rate(attacker):
    base_crit = attacker.get('accuracy', 0) * 0.05 + attacker.get('sense', 0) * 0.08
    # 渐近增长，上限50%
    return min(base_crit, 50)
```

## 1.5 战斗奖励

### 1.5.1 奖励计算公式

```python
def calculate_rewards(role, monster, round_count):
    # 1. 基础奖励
    base_exp = 50 + monster['level'] * 10
    base_gold = 10 + monster['level'] * 2
    
    # 2. 层数加成
    layer_multiplier = 1 + 0.05 * monster['layer']
    
    # 3. 气运加成
    luck_multiplier = 1 + role['luck'] * 0.01
    
    # 4. 速度奖励（回合数越少加成越高）
    speed_bonus = max(0, 1 - round_count / 50)
    
    final_exp = int(base_exp * layer_multiplier * luck_multiplier * (1 + speed_bonus))
    final_gold = int(base_gold * layer_multiplier * luck_multiplier)
    
    # 5. 装备掉落
    equip_drop = roll_equipment_drop(monster)
    
    return {
        'exp': final_exp,
        'gold': final_gold,
        'equipment': equip_drop
    }
```

## 1.6 挂机推图规则

### 1.6.1 挂机前置条件

```python
def can_start_afk(role_id, map_id, layer):
    role = get_role(role_id)
    map_config = get_map_config(map_id)
    
    # 1. 检查地图解锁
    if role['realm_level'] < map_config['unlock_realm']:
        return {'can': False, 'msg': '境界不足'}
    
    # 2. 检查层数解锁
    if layer > get_role_map_progress(role_id, map_id) + 1:
        return {'can': False, 'msg': '请先通关前一关卡'}
    
    # 3. 检查战力
    monster = generate_monster(map_id, layer)
    role_dps = calculate_role_dps(role)
    monster_hp = monster['hp']
    
    # 挂机要求：击杀时间<30秒 且 生存率>90%
    kill_time = monster_hp / role_dps
    if kill_time > 30:
        return {'can': False, 'msg': '战力不足，建议手动通关后再挂机'}
    
    return {'can': True}
```

### 1.6.2 挂机收益计算

```python
def calculate_afk_rewards(role_id, afk_record, offline_seconds):
    role = get_role(role_id)
    map_id = afk_record['map_id']
    layer = afk_record['layer']
    
    # 1. 有效时间（上限8小时）
    effective_time = min(offline_seconds, 8 * 3600)
    
    # 2. 每秒收益
    per_second = {
        'exp': calculate_cultivate_speed(role) * 0.8,  # 挂机效率80%
        'gold': calculate_gold_per_second(map_id, layer) * 0.5,  # 挂机金币50%
        'materials': calculate_material_rate(map_id, layer) * 0.3,  # 挂机材料30%
    }
    
    # 3. 总收益
    total = {
        'exp': int(per_second['exp'] * effective_time),
        'gold': int(per_second['gold'] * effective_time),
    }
    
    # 4. 材料随机掉落
    material_count = int(per_second['materials'] * effective_time)
    total['materials'] = roll_materials(material_count, map_id)
    
    # 5. 装备掉落（基于击杀次数估算）
    kill_count = int(effective_time / 10)  # 约10秒击杀一只
    total['equipment'] = roll_equipment_drops(kill_count, map_id, layer)
    
    return total
```

```python
class AfkSettlementJob:
```
