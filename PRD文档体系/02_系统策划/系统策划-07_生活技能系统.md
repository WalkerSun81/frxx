# 系统策划-07_生活技能系统

## 1.1 系统概述

- **系统定位**：资源获取副线，提供材料自给自足能力
- **核心循环**：采集材料 → 制作装备/丹药 → 提升战力/加速修炼
- **设计目标**：丰富游戏内容，提供离线收益来源

### 1.3.3 采集算法

```python
def collect(skill_type, role_id):
    skill = get_life_skill(role_id, skill_type)
    
    # 1. 检查每日次数限制
    if skill['daily_count'] >= DAILY_LIMIT[skill['level']]:
        return {'success': False, 'msg': '今日采集次数已用完'}
    
    # 2. 扣除体力（如有）
    if not deduct_stamina(10):
        return {'success': False, 'msg': '体力不足'}
    
    # 3. 计算产出
    items = []
    for material_id, probability in get_material_pool(skill['level']):
        if random() < probability:
            count = random.randint(1, 3)
            items.append({'id': material_id, 'count': count})
    
    # 4. 增加经验
    exp_gain = 5
    level_up_result = add_skill_exp(skill, exp_gain)
    
    # 5. 记录每日次数
    increment_daily_count(skill)
    
    return {
        'success': True,
        'items': items,
        'exp_gain': exp_gain,
        'level_up': level_up_result
    }
```

## 1.5 制作算法

```python
def craft(recipe_id, role_id):
    recipe = get_recipe(recipe_id)
    skill = get_life_skill(role_id, recipe['skill_type'])
    
    # 1. 检查专精等级
    if skill['level'] < recipe['min_level']:
        return {'success': False, 'msg': f'需要{recipe["skill_type"]}Lv.{recipe["min_level"]}'}
    
    # 2. 检查材料
    for material_id, count in recipe['materials']:
        if get_material_count(role_id, material_id) < count:
            return {'success': False, 'msg': f'材料不足：{material_id}'}
    
    # 3. 扣除材料
    for material_id, count in recipe['materials']:
        deduct_material(role_id, material_id, count)
    
    # 4. 生成产物
    product = generate_product(recipe)
    
    # 5. 经验加成（高品质额外经验）
    exp_gain = 10
    if product['quality'] >= 3:  # 蓝色及以上
        exp_gain += 20
    
    add_skill_exp(skill, exp_gain)
    
    return {
        'success': True,
        'product': product,
        'exp_gain': exp_gain
    }

def generate_product(recipe):
    # 品质判定
    quality_roll = random()
    if quality_roll < 0.4:  # 40% 普通
        quality = 1
    elif quality_roll < 0.7:  # 30% 绿
        quality = 2
    elif quality_roll < 0.85:  # 15% 蓝
        quality = 3
    elif quality_roll < 0.95:  # 10% 紫
        quality = 4
    else:  # 5% 橙
        quality = 5
    
    # 基础属性×品质倍率
    base_attrs = recipe['base_attrs']
    final_attrs = {k: int(v * QUALITY_MULTIPLIER[quality]) for k, v in base_attrs.items()}
    
    return {
        'type': recipe['product_type'],
        'quality': quality,
        'attrs': final_attrs
    }

# 品质属性倍率表
QUALITY_MULTIPLIER = {
    1: 1.0,   # 普通
    2: 1.2,   # 绿色
    3: 1.5,   # 蓝色
    4: 2.0,   # 紫色
    5: 2.5,   # 橙色
}
```

## 1.6 专精升级规则

```python
def add_skill_exp(skill, exp):
    skill['exp'] += exp
    new_level = skill['level']
    
    # 检查升级
    while new_level < 5 and skill['exp'] >= LEVEL_UP_REQUIRE[skill['level']]:
        skill['exp'] -= LEVEL_UP_REQUIRE[new_level]
        new_level += 1
    
    if new_level > skill['level']:
        skill['level'] = new_level
        save_skill(skill)
        return {'level_up': True, 'new_level': new_level}
    
    save_skill(skill)
    return {'level_up': False}

LEVEL_UP_REQUIRE = {
    1: 100,    # Lv.1 → Lv.2
    2: 500,    # Lv.2 → Lv.3
    3: 2000,   # Lv.3 → Lv.4
    4: 8000,   # Lv.4 → Lv.5
}
```
